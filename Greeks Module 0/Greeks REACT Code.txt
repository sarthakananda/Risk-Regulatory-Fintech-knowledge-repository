const App = () => {
  const [currentIdx, setCurrentIdx] = useState(0);
  const [isPlaying, setIsPlaying] = useState(false);
  const [isNarrating, setIsNarrating] = useState(false);
  const [progress, setProgress] = useState(0);
  const [speed, setSpeed] = useState(1.0);
  
  const currentSlide = SLIDES[currentIdx];
  const synthesisRef = useRef(window.speechSynthesis);
  const utteranceRef = useRef(null);
  const progressIntervalRef = useRef(null);

  const stopAudio = useCallback(() => {
    if (synthesisRef.current) {
      synthesisRef.current.cancel();
    }
    if (progressIntervalRef.current) {
      clearInterval(progressIntervalRef.current);
    }
    setIsNarrating(false);
    setProgress(0);
  }, []);

  const handleNext = useCallback(() => {
    stopAudio();
    setCurrentIdx(prev => {
      if (prev < SLIDES.length - 1) return prev + 1;
      setIsPlaying(false);
      return 0;
    });
  }, [stopAudio]);

  const handlePrev = useCallback(() => {
    stopAudio();
    setCurrentIdx(prev => Math.max(0, prev - 1));
  }, [stopAudio]);

  useEffect(() => {
    if (!isPlaying) {
      stopAudio();
      return;
    }

    const text = currentSlide.narration;
    const utterance = new SpeechSynthesisUtterance(text);
    
    // Setup utterance
    utterance.rate = speed;
    utterance.pitch = 1.0;
    utteranceRef.current = utterance;

    // We estimate duration because Web Speech API doesn't provide real-time progress on some browsers
    const estimatedDuration = (text.split(' ').length / (150 * speed)) * 60 * 1000;
    let startTime = Date.now();

    utterance.onstart = () => {
      setIsNarrating(true);
      progressIntervalRef.current = setInterval(() => {
        const elapsed = Date.now() - startTime;
        const p = Math.min((elapsed / estimatedDuration) * 100, 99);
        setProgress(p);
      }, 100);
    };

    utterance.onend = () => {
      setIsNarrating(false);
      setProgress(100);
      clearInterval(progressIntervalRef.current);
      if (isPlaying) {
        setTimeout(handleNext, 800);
      }
    };

    utterance.onerror = () => {
      setIsNarrating(false);
      if (isPlaying) setTimeout(handleNext, 2000);
    };

    synthesisRef.current.speak(utterance);

    return () => stopAudio();
  }, [currentIdx, isPlaying, handleNext, stopAudio, speed]);

  const togglePlay = () => setIsPlaying(!isPlaying);

  return (
    <div className="min-h-screen bg-black flex flex-col items-center justify-center p-2 font-sans select-none overflow-hidden">
      
      <div className={`relative max-w-7xl w-full h-[700px] rounded-[1rem] overflow-hidden shadow-2xl border border-white/5 transition-all duration-1000 ${currentSlide.bg}`}>
        
        {/* Progress Bar */}
        <div className="absolute top-0 left-0 w-full h-[2px] bg-white/5 z-50">
          <div className="h-full bg-blue-500 transition-all duration-150 ease-linear shadow-[0_0_10px_#3b82f6]" style={{ width: `${progress}%` }} />
        </div>

        {/* Content Layer */}
        <div className="relative h-full flex flex-col p-12 pb-20 text-white">
          
          <div className="flex justify-between items-start mb-8 z-10">
            <div className="flex items-center gap-4">
              <div className="p-3 bg-white/10 rounded-2xl border border-white/10 backdrop-blur-xl shadow-inner">
                <Binary className="w-6 h-6 text-blue-400" />
              </div>
              <div className="flex flex-col">
                <span className="text-[14px] font-black tracking-[0.2em] uppercase text-white/90">Greeks & Sensitivities</span>
                <span className="text-[10px] font-bold opacity-50 uppercase tracking-[0.1em]">Practitioners knowledge repository volume 0</span>
              </div>
            </div>
            
            <div className="bg-black/40 px-4 py-2 rounded-full border border-white/10 backdrop-blur-md flex items-center gap-4">
              <span className="text-[10px] font-mono font-bold opacity-30 uppercase">{currentIdx + 1}/{SLIDES.length}</span>
            </div>
          </div>

          <div className="flex-1 flex flex-col md:flex-row items-center gap-12 z-10 overflow-visible">
            <div className="flex-1 space-y-8 max-w-xl">
              <div key={`title-${currentIdx}`} className="animate-in fade-in slide-in-from-left-8 duration-700">
                <h1 className="text-7xl font-black tracking-tighter mb-4 leading-[0.8] drop-shadow-2xl">{currentSlide.title}</h1>
                <p className="text-2xl font-bold text-white/30 tracking-tight italic flex items-center gap-4">
                  <ChevronRight className="w-5 h-5 text-blue-500" />
                  {currentSlide.subtitle}
                </p>
              </div>

              {currentSlide.scientific && (
                <div key={`meta-${currentIdx}`} className="flex flex-col gap-3 animate-in fade-in zoom-in-95 duration-700 delay-150">
                  <div className="flex items-center gap-4 bg-blue-500/10 border border-blue-500/20 px-4 py-2 rounded-xl">
                    <BookOpen className="w-4 h-4 text-blue-400" />
                    <span className="text-xs font-bold uppercase tracking-widest text-blue-300 opacity-60">Scientific:</span>
                    <span className="font-mono text-lg text-white">${currentSlide.scientific}$</span>
                  </div>
                  <div className="flex items-center gap-4 bg-white/5 border border-white/10 px-4 py-2 rounded-xl">
                    <BarChart3 className="w-4 h-4 text-purple-400" />
                    <span className="text-xs font-bold uppercase tracking-widest text-purple-300 opacity-60">Instrument:</span>
                    <span className="text-sm font-medium text-white/80">{currentSlide.example}</span>
                  </div>
                </div>
              )}

              <div key={`desc-${currentIdx}`} className="p-8 bg-white/5 backdrop-blur-3xl rounded-[2.5rem] border border-white/10 shadow-inner animate-in fade-in zoom-in-95 duration-700 delay-200">
                <p className="text-2xl leading-relaxed font-medium opacity-80">{currentSlide.content}</p>
              </div>
              
              {currentIdx === 0 && !isPlaying && (
                <button onClick={togglePlay} className="px-12 py-6 rounded-full bg-white text-black font-black uppercase tracking-widest transition-all shadow-2xl flex items-center gap-4 group hover:scale-105 active:scale-95">
                  <Play className="w-6 h-6 fill-black" />
                  Start Mastery Experience
                </button>
              )}
            </div>

            <div key={`visual-${currentIdx}`} className="flex-1 h-full flex justify-center items-center animate-in fade-in zoom-in duration-700 relative overflow-visible">
               <OptionMobile animation={currentSlide.visual} />
            </div>
          </div>

          <div className="absolute bottom-0 left-0 right-0 h-16 flex items-center justify-between px-12 bg-gradient-to-t from-black/40 to-transparent backdrop-blur-[1px] transition-all duration-300 opacity-30 hover:opacity-100 z-[60]">
            <div className="flex items-center gap-10">
              <button onClick={handlePrev} className="p-2 hover:bg-white/10 rounded-full transition-all text-white/40 hover:text-white"><SkipBack className="w-5 h-5" /></button>
              <button onClick={togglePlay} className="flex items-center justify-center transition-all hover:scale-110 active:scale-90">
                {isPlaying ? <Pause className="w-6 h-6 text-white/60" /> : <Play className="w-6 h-6 fill-white text-white/60" />}
              </button>
              <button onClick={handleNext} className="p-2 hover:bg-white/10 rounded-full transition-all text-white/40 hover:text-white"><SkipForward className="w-5 h-5" /></button>
            </div>

            <div className="flex-1 px-20 flex flex-col justify-center">
               <div className="h-[1px] w-full bg-white/10 rounded-full overflow-hidden">
                  <div className="h-full bg-white/40 transition-all duration-300" style={{ width: `${(currentIdx + 1) * (100 / SLIDES.length)}%` }} />
               </div>
            </div>

            <div className="flex items-center gap-10">
               <div className="flex gap-2 bg-white/5 p-1 rounded-lg border border-white/10">
                 {[0.5, 0.8, 1.0, 1.2].map(s => (
                   <button key={s} onClick={() => { setSpeed(s); if (isPlaying) stopAudio(); }} className={`text-[9px] font-black px-2 py-0.5 rounded transition-all ${speed === s ? 'bg-white text-black' : 'text-white/30 hover:text-white/60'}`}>{s}x</button>
                 ))}
               </div>
               <div className={`flex items-center gap-3 transition-all ${isNarrating ? 'text-blue-400 opacity-100' : 'text-white/20'}`}>
                 {isNarrating ? <Volume2 className="w-5 h-5 animate-pulse" /> : <VolumeX className="w-5 h-5" />}
               </div>
               <button onClick={() => { stopAudio(); setCurrentIdx(0); setProgress(0); setIsPlaying(true); }} className="p-2 text-white/20 hover:text-white transition-all hover:rotate-180 duration-500"><RefreshCw className="w-5 h-5" /></button>
            </div>
          </div>
        </div>
      </div>
      
      <div className="mt-4 flex flex-col items-center">
        <p className="text-[10px] font-black uppercase tracking-[0.6em] text-white/20">Module 0 Reference Refinement Engine</p>
        <p className="text-[9px] font-bold text-white/10 mt-1 uppercase tracking-widest italic">Author: Sarthaka Nanda in collaboration with Gemini Canvas</p>
      </div>
    </div>
  );
};

export default App;
